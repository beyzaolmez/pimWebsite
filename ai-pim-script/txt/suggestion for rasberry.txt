import requests

# Specify the URL of the server
server_url = "http://<server_ip>:8000"  # Replace <server_ip> with the actual IP address of the server

# Specify the path to the audio file on the Raspberry Pi
audio_file_path = "/path/to/audio_file.wav"  # Replace with the actual path to the audio file

# Read the audio file as binary data
with open(audio_file_path, "rb") as audio_file:
    audio_data = audio_file.read()

# Send the audio file to the server
response = requests.post(server_url, data=audio_data)

# Check the response status
if response.status_code == 200:
    print("Audio file sent successfully")
else:
    print(f"Failed to send audio file. Status code: {response.status_code}")
